\paragraph{Question}
A flashlight is aimed at an aircraft flying at an altitude $h$ with a constant velocity $\vec{v}$.
Its beam continues to point directly towards the position of the aircraft as it moves. At $t = 0$,
its instantaneous angular velocity is $\omega_0$. Express the instantaneous angular velocity $\omega$
of the rotation of the flashlight's beam as a function of time.

\begin{center}
\begin{tikzpicture}[scale=1.5]
	\coordinate [label=right:$x$]		(X)		at	(3,0);
	\coordinate [label=above:$y$]		(Y)		at	(0,4);

	\coordinate							(O)		at	(0,0);

											node[midway, above] {$vt$};
	\draw[dotted, very thick, yellow] (O) -- ($(A) + (1.732050807568877,0)$);
	\draw[dotted, yellow] (O) --++ (60:4);
	\draw[-latex, very thick, yellow] (O) --++ (60:2) node[midway, above, sloped, black] 	{beam};
	
	\draw[latex-] (O) ++ (60:6mm) arc (60:90:6mm);
	\node at ($ (O) + (1mm, 4mm)$) {$\theta$};
	
	\node[left] at (O) {$0$};
	\node[left] at (A) {$h$};
	
	
\end{tikzpicture}

\paragraph{Solution}
We have
\begin{align*}
	\tan\theta			\;&=\;	\frac{vt}{h}		\\
	\theta				\;&=\;	\tan^{-1}\frac{vt}{h}	\\
	\frac{d\theta}{dt}	\;&=\;	\omega(t)	\;=\;	\frac{1}{1 + (\frac{vt}{h})^2} \cdot \frac{v}{h}		\tag{1}
\end{align*}

Note that $\omega(0) = \omega_0 = \frac{v}{h}$. Thus
\begin{align*}
	\boxed{\omega(t)	\;=\;	\dfrac{\omega_0}{1 + \omega_0^2 t^2}}		\tag{2}
\end{align*}
